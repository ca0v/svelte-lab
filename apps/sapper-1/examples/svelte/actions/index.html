function action1(node) { node.classList.add("action"); node.style.background = "black"; node.tabIndex = -1; node.addEventListener("focus", () => { node.style.color = "red"; }); node.addEventListener("focusout", () => { node.style.color = ""; }); }

function action2(node, title) { node.title = title; return { update: title => { node.title = title; } }; }
